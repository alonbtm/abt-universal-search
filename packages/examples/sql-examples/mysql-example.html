MySQL Database Schema

MySQL setup with FULLTEXT indexing and spatial search capabilities:

-- Create database and table with FULLTEXT search support CREATE DATABASE search_demo CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci; USE search_demo; CREATE TABLE products ( id INT PRIMARY KEY AUTO_INCREMENT, name VARCHAR(255) NOT NULL, description TEXT, category VARCHAR(100), tags JSON, price DECIMAL(10,2), location POINT SRID 4326, -- Spatial data for location-based search search_vector TEXT, -- Computed search field created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP, -- FULLTEXT indexes for different search strategies FULLTEXT KEY ft_name (name), FULLTEXT KEY ft_description (description), FULLTEXT KEY ft_combined (name, description), -- Spatial index for location search SPATIAL KEY idx_location (location), -- Regular indexes for filtering KEY idx_category (category), KEY idx_price (price), KEY idx_created_at (created_at) ) ENGINE=InnoDB; -- Sample data insertion INSERT INTO products (name, description, category, tags, price, location, search_vector) VALUES ('MacBook Pro M3', 'High-performance laptop for developers and creators', 'Electronics', '["laptop", "apple", "programming", "design"]', 2999.00, ST_GeomFromText('POINT(-122.4194 37.7749)', 4326), -- San Francisco 'MacBook Pro M3 High-performance laptop for developers and creators Electronics'), ('iPhone 15 Pro', 'Latest smartphone with advanced camera system', 'Electronics', '["smartphone", "apple", "photography", "mobile"]', 1199.00, ST_GeomFromText('POINT(-122.4194 37.7749)', 4326), 'iPhone 15 Pro Latest smartphone with advanced camera system Electronics'), ('Dell XPS 13', 'Ultrabook with excellent display and portability', 'Electronics', '["laptop", "dell", "ultrabook", "portable"]', 1299.00, ST_GeomFromText('POINT(-87.6298 41.8781)', 4326), -- Chicago 'Dell XPS 13 Ultrabook with excellent display and portability Electronics'); -- Create FULLTEXT index on computed search_vector ALTER TABLE products ADD FULLTEXT KEY ft_search_vector (search_vector);
MySQL Performance Tip: FULLTEXT indexes in MySQL use different ranking algorithms than PostgreSQL. Use MATCH() AGAINST() with IN BOOLEAN MODE for complex queries and relevance scoring.

MySQL FULLTEXT search with Boolean mode and relevance scoring:

-- Basic FULLTEXT search with relevance scoring SELECT id, name, description, MATCH(name, description) AGAINST(? IN BOOLEAN MODE) as relevance_score FROM products WHERE MATCH(name, description) AGAINST(? IN BOOLEAN MODE) ORDER BY relevance_score DESC, created_at DESC LIMIT ?; -- Boolean operators in search -- +word (must contain), -word (must not contain), "phrase" (exact phrase) SELECT * FROM products WHERE MATCH(search_vector) AGAINST('+laptop -apple "high performance"' IN BOOLEAN MODE); -- Natural language search with minimum score threshold SELECT *, MATCH(search_vector) AGAINST(? IN NATURAL LANGUAGE MODE) as score FROM products WHERE MATCH(search_vector) AGAINST(? IN NATURAL LANGUAGE MODE) HAVING score > 0.5 ORDER BY score DESC;

MySQL spatial search for location-based queries:

-- Find products within radius (in meters) SELECT id, name, price, ST_Distance_Sphere(location, ST_GeomFromText('POINT(-122.4194 37.7749)', 4326)) as distance_meters FROM products WHERE ST_Distance_Sphere(location, ST_GeomFromText('POINT(? ?)', 4326)) <= ? ORDER BY distance_meters ASC LIMIT ?; -- Bounding box search for rectangular areas SELECT id, name, location FROM products WHERE ST_Within(location, ST_GeomFromText('POLYGON(( -122.5 37.7, -122.3 37.7, -122.3 37.8, -122.5 37.8, -122.5 37.7))', 4326) ); -- Combined text and location search SELECT *, MATCH(search_vector) AGAINST(? IN BOOLEAN MODE) as text_score, ST_Distance_Sphere(location, ST_GeomFromText('POINT(? ?)', 4326)) as distance FROM products WHERE MATCH(search_vector) AGAINST(? IN BOOLEAN MODE) AND ST_Distance_Sphere(location, ST_GeomFromText('POINT(? ?)', 4326)) <= 50000 ORDER BY (text_score * 0.7) + ((50000 - distance) / 50000 * 0.3) DESC;

MySQL JSON column search and manipulation:

-- Search within JSON arrays (tags) SELECT id, name, tags, JSON_EXTRACT(tags, '$[*]') as tag_list FROM products WHERE JSON_CONTAINS(tags, JSON_QUOTE(?)) OR JSON_SEARCH(tags, 'one', ?) IS NOT NULL; -- Complex JSON path queries SELECT id, name, JSON_EXTRACT(tags, '$[0]') as first_tag FROM products WHERE JSON_LENGTH(tags) > 2 AND JSON_EXTRACT(tags, '$[*]') LIKE '%laptop%'; -- Update JSON fields UPDATE products SET tags = JSON_ARRAY_APPEND(tags, '$', ?) WHERE id = ?; -- Search with JSON conditions SELECT * FROM products WHERE JSON_CONTAINS(tags, '["apple", "laptop"]') AND MATCH(search_vector) AGAINST(? IN BOOLEAN MODE);

Combined search patterns with multiple MySQL features:

-- Multi-dimensional search: text + location + price + JSON SELECT id, name, description, price, MATCH(search_vector) AGAINST(? IN BOOLEAN MODE) as text_relevance, ST_Distance_Sphere(location, ST_GeomFromText('POINT(? ?)', 4326)) as distance, CASE WHEN price BETWEEN ? AND ? THEN 1.0 ELSE 0.5 END as price_score, CASE WHEN JSON_CONTAINS(tags, JSON_QUOTE(?)) THEN 1.0 ELSE 0.0 END as tag_match FROM products WHERE MATCH(search_vector) AGAINST(? IN BOOLEAN MODE) AND ST_Distance_Sphere(location, ST_GeomFromText('POINT(? ?)', 4326)) <= ? AND price >= ? ORDER BY (text_relevance * 0.4) + ((50000 - distance) / 50000 * 0.3) + (price_score * 0.2) + (tag_match * 0.1) DESC LIMIT ?; -- Faceted search with aggregations SELECT category, COUNT(*) as count, AVG(price) as avg_price, MIN(price) as min_price, MAX(price) as max_price FROM products WHERE MATCH(search_vector) AGAINST(? IN BOOLEAN MODE) GROUP BY category ORDER BY count DESC;

MySQL Performance Optimization

MySQL-specific optimization strategies:

  • FULLTEXT Index Types: Use different ft_min_word_len and ft_max_word_len settings
  • Query Cache: Enable query_cache_type for repeated searches
  • Index Merge: MySQL can combine multiple indexes automatically
  • Spatial Optimization: Use appropriate SRID for your geographic data
-- Performance analysis queries EXPLAIN SELECT * FROM products WHERE MATCH(search_vector) AGAINST('laptop' IN BOOLEAN MODE); -- Index usage statistics SELECT TABLE_NAME, INDEX_NAME, SEQ_IN_INDEX, COLUMN_NAME, CARDINALITY FROM information_schema.STATISTICS WHERE TABLE_SCHEMA = 'search_demo' AND TABLE_NAME = 'products'; -- Query performance monitoring SELECT DIGEST_TEXT, COUNT_STAR, AVG_TIMER_WAIT/1000000000 as avg_time_sec FROM performance_schema.events_statements_summary_by_digest WHERE DIGEST_TEXT LIKE '%products%' ORDER BY COUNT_STAR DESC;
