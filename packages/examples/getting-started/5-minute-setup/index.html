import { UniversalSearch, SearchConfig, SearchItem } from './src/index'; // Define your data type interface Product extends SearchItem { name: string; category: string; price: number; description?: string; } // Configure the search const config: SearchConfig = { data: productData, searchKeys: ['name', 'category', 'description'], placeholder: 'Search products...', maxResults: 10, debounceMs: 300, fuzzySearch: true, highlight: true, onSearch: (query: string, results: Product[]) => { console.log(`Found ${results.length} products for "${query}"`); }, onSelect: (product: Product) => { console.log('Selected:', product.name); } }; // Initialize and mount const search = new UniversalSearch(config); search.mount('#search-container');
